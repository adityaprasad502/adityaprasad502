fun cuteMsg() {
    val mt = listOf(65, 68, 73, 84, 89, 65, 32, 80, 82, 65, 83, 65, 68, 32, 83)

    while (true) {
        for (i in 0..13) {
            println("--".repeat(i) + mt[i].toChar())
            Thread.sleep(96)
        }
        for (i in 14 downTo 1) {
            println("--".repeat(i) + mt[i].toChar())
            Thread.sleep(96)
        }
    }
}

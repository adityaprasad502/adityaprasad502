    while (true) {
        for (i in 0..13) {
            println("--".repeat(i) + mt[i].toChar())
            Thread.sleep(96)
        }
        for (i in 14 downTo 1) {
            println("--".repeat(i) + mt[i].toChar())
            Thread.sleep(96)
        }
    }
}
